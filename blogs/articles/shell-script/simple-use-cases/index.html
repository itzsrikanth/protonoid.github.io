To convert nunjuck template without space

#!/bin/bash

find . -name index.html | while read line; do
    perl -pi -e 's/{(%|#)(?=[^-])/{\1-/g' $line
    perl -pi -e 's/(?<=[^-])(%|#)}/-\1}/g' $line
done
